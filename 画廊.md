```dataviewjs

// ====== 全局配置 ======
const TYPES = ["novel", "anime", "comic", "movie"];
const container = dv.container;

// ====== 获取所有媒体数据 ======
let pages = dv.pages()
    .where(p => p.type && TYPES.includes(p.type));

// ====== 状态集合 ======
const statuses = [...new Set(pages.map(p => p.status).filter(Boolean))];

// ====== 当前筛选状态 ======
let current = {
    type: "all",
    status: "all",
    rating: 0
};

// ====== UI容器 ======
const filterBar = container.createEl("div", { cls: "filter-bar" });
const grid = container.createEl("div", { cls: "media-grid" });

// ====== 按钮生成函数 ======
function createButton(text, key, value) {
    let btn = filterBar.createEl("button", {
        text: text,
        cls: "filter-btn"
    });

    btn.onclick = () => {
        current[key] = value;
        render();
    };
}

// ====== 生成筛选按钮 ======

// 类型
createButton("全部", "type", "all");
TYPES.forEach(t => createButton(t, "type", t));

// 状态
createButton("全部状态", "status", "all");
statuses.forEach(s => createButton(s, "status", s));

// 评分筛选
[0, 2, 4, 6, 8].forEach(r => {
    createButton(`≥${r}`, "rating", r);
});

// ====== 渲染函数 ======
function render() {
    grid.innerHTML = "";

    let filtered = pages
        .where(p => current.type === "all" || p.type === current.type)
        .where(p => current.status === "all" || p.status === current.status)
        .where(p => !p.rating || p.rating >= current.rating)
        .sort(p => p.rating, 'desc');

    filtered.forEach(p => {
        let card = grid.createEl("div", { cls: "media-card" });

        // 封面
        if (p.cover) {
            card.createEl("img", {
                attr: { src: p.cover },
                cls: "media-cover"
            });
        }

        // 标题
        card.createEl("div", {
            text: p.file.name,
            cls: "media-title"
        });

        // 类型
        card.createEl("div", {
            text: p.type,
            cls: "media-type"
        });

        // 作者
        if (p.author) {
            card.createEl("div", {
                text: p.author,
                cls: "media-author"
            });
        }

        // 评分
        if (p.rating) {
            let stars = "⭐".repeat(Math.round(p.rating / 2));
            card.createEl("div", {
                text: stars,
                cls: "media-rating"
            });
        }
    });
}

// ====== 初始渲染 ======
render();

// ====== 样式 ======
const style = document.createElement("style");
style.textContent = `

    margin-bottom: 16px;
}

.filter-btn {

    padding: 4px 10px;
    border-radius: 8px;
    border: none;
    background: var(--interactive-normal);
    cursor: pointer;
}

.media-grid {
    display: grid;
    grid-template-columns: repeat(auto-fill, minmax(160px, 1fr));
    gap: 16px;
}

.media-card {
    background: var(--background-secondary);
    border-radius: 12px;
    padding: 10px;
    text-align: center;
    transition: 0.2s;
}

.media-card:hover {
    transform: translateY(-5px);
}

.media-cover {
    width: 100%;
    height: 220px;
    object-fit: cover;
    border-radius: 8px;
}

.media-title {
    font-weight: bold;
    margin-top: 6px;
}

.media-type {
    font-size: 12px;
    color: gray;
}

.media-author {
    font-size: 12px;
    color: gray;
}

.media-rating {
    margin-top: 4px;
}
`;
document.head.appendChild(style);
```
