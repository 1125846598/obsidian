// ================== 配置 ==================
const TYPES = ["book", "movie", "teleplay", "anime", "comic", "game"];
const NS = "mlib";

let state = {
    type: "all",
    status: "all",
    rating: "all"
};

// ================== 容器 ==================
const container = dv.container;
container.innerHTML = "";
container.classList.add(`${NS}-root`);

const filter = container.createEl("div");
filter.className = `${NS}-filter`;

const grid = container.createEl("div");
grid.className = `${NS}-grid`;

// ================== 数据 ==================
let pages = dv.pages().where(p => p.type);

// ================== 按钮 ==================
function btn(parent, text, key, value) {
    let b = parent.createEl("button", { text });
    b.className = `${NS}-btn`;
    if (state[key] === value) b.classList.add(`${NS}-btn--active`);
    b.onclick = () => {
        state[key] = value;
        render();
    };
}

// ================== 封面 ==================
function getCover(p) {
    if (!p.cover) return null;

    let c = p.cover;
    if (Array.isArray(c)) c = c[0];

    let file = null;
    if (c?.path) {
        file = app.metadataCache.getFirstLinkpathDest(c.path, p.file.path);
    } else if (typeof c === "string") {
        file = app.metadataCache.getFirstLinkpathDest(c, p.file.path);
    }

    if (file) return app.vault.getResourcePath(file);
    return null;
}

// ================== 渲染 ==================
function render() {
    filter.innerHTML = "";
    grid.innerHTML = "";

    // ===== 类型 =====
    const typeRow = filter.createEl("div");
    typeRow.className = `${NS}-filter-row`;
    btn(typeRow, "全部", "type", "all");
    TYPES.forEach(t => btn(typeRow, t, "type", t));

    // ===== 状态 =====
    const statusRow = filter.createEl("div");
    statusRow.className = `${NS}-filter-row`;
    btn(statusRow, "全部状态", "status", "all");
    btn(statusRow, "✓ 已完成", "status", "completed");
    btn(statusRow, "▷ 进行中", "status", "ongoing");

    // ===== 评分 =====
    const ratingRow = filter.createEl("div");
    ratingRow.className = `${NS}-filter-row`;
    btn(ratingRow, "全部评分", "rating", "all");
    [4, 6, 8, 10].forEach(r => btn(ratingRow, "⭐".repeat(r / 2), "rating", r));

    // ================== 过滤 ==================
    let list = pages
        .where(p => state.type === "all" || p.type === state.type)
        .where(p => state.status === "all" || (p.status ?? "") === state.status)
        .where(p => state.rating === "all" || (p.rating && Number(p.rating) >= Number(state.rating)));

    // ================== 卡片 ==================
    list.forEach(p => {
        let card = grid.createEl("div");
        card.className = `${NS}-card`;

        let wrap = card.createEl("div");
        wrap.className = `${NS}-wrap`;

        wrap.style.cursor = "pointer";
        wrap.onclick = () => {
            app.workspace.openLinkText(p.file.path, "", false);
        };

        let url = getCover(p);

        if (url) {
            let img = wrap.createEl("img");
            img.className = `${NS}-img`;
            img.src = url;
        } else {
            let placeholder = wrap.createEl("div");
            placeholder.className = `${NS}-placeholder`;
            placeholder.innerText = p.file.name.charAt(0);
        }

        // ===== 渐变 =====
        let gradient = wrap.createEl("div");
        gradient.className = `${NS}-gradient`;

        // ===== 类型 =====
        let tag = wrap.createEl("div");
        tag.className = `${NS}-tag`;
        tag.innerText = p.type;

        // ===== 评分 =====
        if (p.rating) {
            let rating = wrap.createEl("div");
            rating.className = `${NS}-rating`;
            rating.innerText = "⭐".repeat(Math.round(p.rating / 2));
        }

        // ===== 标题 =====
        let title = wrap.createEl("div");
        title.className = `${NS}-title`;
        title.innerText = p.file.name;

        // ================== 状态角标（重点修改） ==================

        if ((p.status ?? "").toLowerCase().trim() === "completed") {
            let done = wrap.createEl("div");
            done.className = `${NS}-done`;
            done.innerText = "✓";
        }

        if ((p.status ?? "").toLowerCase().trim() === "ongoing") {
            let doing = wrap.createEl("div");
            doing.className = `${NS}-done`;
            doing.style.background = "#f59e0b";
            doing.innerText = "▷";
        }
    });
}

render();
