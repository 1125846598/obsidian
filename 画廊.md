// ====== 配置 ======
const TYPES = ["novel", "anime", "comic", "movie"];
const STATUS = ["all", "completed", "doing", "planned"];

let current = {
    type: "all",
    status: "all",
    rating: 0
};

const container = dv.container;
const filterBar = container.createEl("div", { cls: "filter-bar" });
const grid = container.createEl("div", { cls: "media-grid" });

// ====== 数据 ======
let pages = dv.pages()
    .where(p => p.type && TYPES.includes(p.type));

// ====== 工具函数 ======
function createGroup(title) {
    let g = filterBar.createEl("div", { cls: "filter-group" });
    g.createEl("span", { text: title, cls: "filter-title" });
    return g;
}

function createBtn(parent, text, key, value) {
    let btn = parent.createEl("button", {
        text: text,
        cls: "filter-btn"
    });

    btn.onclick = () => {
        current[key] = value;
        render();
    };
}

// ====== 筛选UI ======

// 类型
let g1 = createGroup("类型:");
createBtn(g1, "全部", "type", "all");
TYPES.forEach(t => createBtn(g1, t, "type", t));

// 状态
let g2 = createGroup("状态:");
createBtn(g2, "全部", "status", "all");
["completed", "doing", "planned"].forEach(s => {
    createBtn(g2, s, "status", s);
});

// 评分（星级）
let g3 = createGroup("评分:");
createBtn(g3, "全部", "rating", 0);
[2,4,6,8].forEach(r => {
    createBtn(g3, "⭐".repeat(r/2), "rating", r);
});

// ====== 渲染 ======
function render() {
    grid.innerHTML = "";

    let filtered = pages
        .where(p => current.type === "all" || p.type === current.type)
        .where(p => current.status === "all" || p.status === current.status)
        .where(p => !p.rating || p.rating >= current.rating)
        .sort(p => p.rating, 'desc');

    filtered.forEach(p => {
        let card = grid.createEl("div", { cls: "card" });

        // ===== 封面 =====
        let coverWrap = card.createEl("div", { cls: "cover-wrap" });

        if (p.cover) {
            coverWrap.createEl("img", {
                attr: { src: p.cover },
                cls: "cover"
            });
        }

        // ===== 左上角：状态 =====
        if (p.status === "completed") {
            coverWrap.createEl("div", { text: "✓", cls: "status done" });
        } else if (p.status === "doing") {
            coverWrap.createEl("div", { text: "▶", cls: "status doing" });
        }

        // ===== 右上角：类型 =====
        coverWrap.createEl("div", {
            text: p.type,
            cls: "tag"
        });

        // ===== 底部评分（叠加）=====
        if (p.rating) {
            let stars = "⭐".repeat(Math.round(p.rating / 2));
            coverWrap.createEl("div", {
                text: stars,
                cls: "rating"
            });
        }

        // ===== 标题 =====
        card.createEl("div", {
            text: p.file.name,
            cls: "title"
        });

        // 作者
        if (p.author) {
            card.createEl("div", {
                text: p.author,
                cls: "author"
            });
        }
    });
}

render();
