```dataviewjs
const container = dv.container;

// ====== 可修改配置 ======
const PATH = '"2026上半年体验作品/小说"'; // 你的小说路径
const CARD_WIDTH = "180px";

// ====== 获取数据 ======
let pages = dv.pages(PATH)
    .where(p => p.type && p.type === "novel");

// ====== 排序（默认按评分） ======
pages = pages.sort(p => p.rating, 'desc');

// ====== 创建卡片容器 ======
let grid = container.createEl("div", { cls: "book-grid" });

// ====== 渲染函数 ======
function render(pagesList) {
    grid.innerHTML = "";

    pagesList.forEach(p => {
        let card = grid.createEl("div", { cls: "book-card" });

        // 封面
        if (p.cover) {
            card.createEl("img", {
                attr: {
                    src: p.cover,
                },
                cls: "book-cover"
            });
        }

        // 标题
        card.createEl("div", {
            text: p.file.name,
            cls: "book-title"
        });

        // 作者
        if (p.author) {
            card.createEl("div", {
                text: `作者：${p.author}`,
                cls: "book-author"
            });
        }

        // 评分（星星）
        if (p.rating) {
            let stars = "⭐".repeat(Math.round(p.rating / 2));
            card.createEl("div", {
                text: stars,
                cls: "book-rating"
            });
        }
    });
}

// ====== 初始渲染 ======
render(pages);

// ====== 样式 ======
const style = document.createElement("style");
style.textContent = `
.book-grid {
    display: grid;
    grid-template-columns: repeat(auto-fill, minmax(${CARD_WIDTH}, 1fr));
    gap: 16px;
}

.book-card {
    background: var(--background-secondary);
    border-radius: 12px;
    padding: 10px;
    text-align: center;
    transition: 0.2s;
}

.book-card:hover {
    transform: translateY(-4px);
}

.book-cover {
    width: 100%;
    height: 240px;
    object-fit: cover;
    border-radius: 8px;
}

.book-title {
    font-weight: bold;
    margin-top: 8px;
}

.book-author {
    font-size: 12px;
    color: gray;
}

.book-rating {
    margin-top: 4px;
}
`;
document.head.appendChild(style);
```
