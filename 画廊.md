```dataviewjs
// ================== 配置 ==================
const TYPES = ["book", "movie", "teleplay", "anime", "comic"];
const STATUS = ["completed", "doing", "planned"];

// ================== 状态 ==================
let state = {
    type: "all",
    status: "all",
    rating: "all",
    time: "all",
    sort: "rating"
};

// ================== 容器 ==================
const container = dv.container;
const filter = container.createEl("div", { cls: "filter-wrap" });
const grid = container.createEl("div", { cls: "grid" });

// ================== 数据 ==================
let pages = dv.pages().where(p => p.type);

// ================== 工具函数 ==================
function btn(parent, text, key, value) {
    let b = parent.createEl("button", { text, cls: "btn" });

    if (state[key] === value) b.addClass("active");

    b.onclick = () => {
        state[key] = value;
        render();
    };
}

// ================== 筛选 UI ==================
function buildFilters() {
    filter.innerHTML = "";

    // 类型
    let g1 = filter.createEl("div", { cls: "group" });
    btn(g1, "全部", "type", "all");
    TYPES.forEach(t => btn(g1, t, "type", t));

    // 评分
    let g2 = filter.createEl("div", { cls: "group" });
    btn(g2, "全部", "rating", "all");
    [10,8,6,4,2].forEach(r => {
        btn(g2, "⭐".repeat(r/2), "rating", r);
    });

    // 状态
    let g3 = filter.createEl("div", { cls: "group" });
    btn(g3, "全部", "status", "all");
    btn(g3, "已完成", "status", "completed");
    btn(g3, "未完成", "status", "doing");

    // 时间
    let g4 = filter.createEl("div", { cls: "group" });
    btn(g4, "全部", "time", "all");
    btn(g4, "最近一周", "time", "week");
    btn(g4, "最近一月", "time", "month");
    btn(g4, "最近一年", "time", "year");

    // 排序
    let g5 = filter.createEl("div", { cls: "group" });
    g5.createEl("span", { text: "排序方式：" });
    btn(g5, "评分↓", "sort", "rating");
    btn(g5, "标题", "sort", "title");
    btn(g5, "创建时间", "sort", "ctime");
}

// ================== 时间过滤 ==================
function timeFilter(p) {
    if (state.time === "all") return true;

    let now = new Date();
    let t = new Date(p.file.ctime);

    let diff = (now - t) / (1000 * 3600 * 24);

    if (state.time === "week") return diff <= 7;
    if (state.time === "month") return diff <= 30;
    if (state.time === "year") return diff <= 365;
}

// ================== 渲染 ==================
function render() {
    buildFilters();
    grid.innerHTML = "";

    let list = pages
        .where(p => state.type === "all" || p.type === state.type)
        .where(p => state.status === "all" || p.status === state.status)
        .where(p => state.rating === "all" || (p.rating && p.rating >= state.rating))
        .where(p => timeFilter(p));

    // 排序
    if (state.sort === "rating") list = list.sort(p => p.rating, "desc");
    if (state.sort === "title") list = list.sort(p => p.file.name);
    if (state.sort === "ctime") list = list.sort(p => p.file.ctime, "desc");

    list.forEach(p => {
        let card = grid.createEl("div", { cls: "card" });

        let coverWrap = card.createEl("div", { cls: "cover-wrap" });

        // 封面
        if (p.cover) {
            coverWrap.createEl("img", {
                attr: { src: p.cover },
                cls: "cover"
            });
        }

        // 状态 ✔
        if (p.status === "completed") {
            coverWrap.createEl("div", { text: "✓", cls: "badge done" });
        }

        // 类型标签
        coverWrap.createEl("div", {
            text: p.type,
            cls: "badge type"
        });

        // ⭐评分（浮层）
        if (p.rating) {
            coverWrap.createEl("div", {
                text: "⭐".repeat(Math.round(p.rating/2)),
                cls: "rating"
            });
        }

        // 标题
        card.createEl("div", {
            text: p.file.name,
            cls: "title"
        });

        // 作者
        if (p.author) {
            card.createEl("div", {
                text: p.author,
                cls: "author"
            });
        }
    });
}

render();
```
