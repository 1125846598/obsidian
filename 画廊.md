// ================== 配置 ==================
const TYPES = ["book", "movie", "teleplay", "anime", "comic", "game"];

let state = {
    type: "all",
    status: "all",
    rating: "all"
};

// ================== 容器 ==================
const container = dv.container;
container.innerHTML = "";

const filter = container.createEl("div");
const grid = container.createEl("div");

// ================== 数据 ==================
let pages = dv.pages().where(p => p.type);

// ================== 按钮 ==================
function btn(parent, text, key, value) {
    let b = parent.createEl("button", { text });

    if (state[key] === value) b.style.background = "#5b5bf7", b.style.color="white";

    b.onclick = () => {
        state[key] = value;
        render();
    };
}

// ================== 封面修复 ==================
function getCover(p) {
    if (!p.cover) return null;

    let c = p.cover;
    if (Array.isArray(c)) c = c[0];

    let file = null;

    if (c?.path) file = app.metadataCache.getFirstLinkpathDest(c.path, p.file.path);
    else if (typeof c === "string") file = app.metadataCache.getFirstLinkpathDest(c, p.file.path);

    if (file) return app.vault.getResourcePath(file);
    return null;
}

// ================== 渲染 ==================
function render() {
    filter.innerHTML = "";
    grid.innerHTML = "";

    // 筛选栏
    btn(filter, "全部", "type", "all");
    TYPES.forEach(t => btn(filter, t, "type", t));

    btn(filter, "全部评分", "rating", "all");
    [10,8,6,4].forEach(r => btn(filter, "⭐".repeat(r/2), "rating", r));

    let list = pages
        .where(p => state.type==="all" || p.type===state.type)
        .where(p => state.rating==="all" || (p.rating && p.rating>=state.rating));

    list.forEach(p => {

        let card = grid.createEl("div");

        let url = getCover(p);

        // ===== 封面容器 =====
        let wrap = card.createEl("div");

        if (url) {
            wrap.createEl("img", {
                attr: { src: url }
            });
        }

        // ===== 渐变 =====
        let gradient = wrap.createEl("div");

        // ===== 类型 =====
        let tag = wrap.createEl("div");
        tag.innerText = p.type;

        // ===== 评分 =====
        if (p.rating) {
            let rating = wrap.createEl("div");
            rating.innerText = "⭐".repeat(Math.round(p.rating/2));
        }

        // ===== 标题 =====
        let title = wrap.createEl("div");
        title.innerText = p.file.name;

        // ===== 状态 =====
        if (p.status==="completed") {
            let done = wrap.createEl("div");
            done.innerText = "✓";
        }
    });
}

render();
