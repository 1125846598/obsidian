const pages = dv.pages('"小说"').where(p => p.cover && p.rating);
const cols = 4;   // 每行列数，可改为3或5
let html = `<div style="display: grid; grid-template-columns: repeat(${cols}, 1fr); gap: 20px;">`;
for (let p of pages) {
    html += `
        <div style="text-align: center; border: 1px solid #ccc; border-radius: 12px; padding: 12px; background: var(--background-secondary);">
            <img src="${p.cover}" style="width: 100%; aspect-ratio: 2/3; object-fit: cover; border-radius: 8px;">
            <div style="font-weight: bold; margin-top: 8px;">${p.title}</div>
            <div style="font-size: 0.9em; color: var(--text-muted);">${p.author}</div>
            <div>rating ${p.rating}</div>
        </div>
    `;
}
html += `</div>`;
dv.paragraph(html);
