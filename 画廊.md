```dataviewjs
// ============================================================
// 配置区 – 请根据你的实际文件夹修改
// ============================================================
const config = {
    // 你的小说笔记存放在哪个文件夹？（请替换成实际路径）
    folder: '"2026上半年体验作品/小说"',        // 例如: '"小说"' 或 '"Books/Novel"'
    
    // 每页显示的卡片数量
    cardsPerPage: 12,
    
    // 是否显示封面图（没有 cover 的笔记将显示占位符）
    showCover: true,
    
    // 封面图缺失时的占位文字
    placeholderImage: "📖 无封面"
};

// ============================================================
// 以下代码无需修改（除非你懂 JS）
// ============================================================
let books = dv.pages(config.folder)
    .where(p => p.type === 'novel')      // 只显示小说
    .where(p => p.rating)                // 必须有评分
    .sort(p => p.rating, 'desc');        // 按评分从高到低排序

let allData = books.map(p => ({
    file: p.file,
    title: p.title || p.file.name,
    author: p.author || '未知作者',
    rating: p.rating,
    genre: p.genre || '未分类',
    cover: p.cover ? (p.cover.startsWith('http') ? p.cover : app.vault.adapter.getResourcePath(p.cover)) : null,
    date: p.date ? dv.date(p.date) : null
}));

// 获取所有不重复的类型（genre）
let allGenres = [...new Set(allData.map(b => b.genre).filter(g => g))];
allGenres.unshift('全部');

// 状态变量
let currentGenre = '全部';
let minRating = 0;
let maxRating = 10;
let currentDateFilter = 'all';
let customStart = null, customEnd = null;
let currentPage = 1;

function renderGallery() {
    let filtered = [...allData];
    
    // 按类型筛选
    if (currentGenre !== '全部') {
        filtered = filtered.filter(b => b.genre === currentGenre);
    }
    
    // 按评分范围筛选
    filtered = filtered.filter(b => b.rating >= minRating && b.rating <= maxRating);
    
    // 按日期筛选
    const today = new Date();
    if (currentDateFilter === 'week') {
        const weekAgo = new Date(today); weekAgo.setDate(today.getDate() - 7);
        filtered = filtered.filter(b => b.date && b.date >= weekAgo);
    } else if (currentDateFilter === 'month') {
        const monthAgo = new Date(today); monthAgo.setMonth(today.getMonth() - 1);
        filtered = filtered.filter(b => b.date && b.date >= monthAgo);
    } else if (currentDateFilter === 'year') {
        const yearAgo = new Date(today); yearAgo.setFullYear(today.getFullYear() - 1);
        filtered = filtered.filter(b => b.date && b.date >= yearAgo);
    } else if (currentDateFilter === 'custom' && customStart && customEnd) {
        filtered = filtered.filter(b => b.date && b.date >= customStart && b.date <= customEnd);
    }
    
    // 分页
    const totalPages = Math.ceil(filtered.length / config.cardsPerPage);
    const start = (currentPage - 1) * config.cardsPerPage;
    const pageData = filtered.slice(start, start + config.cardsPerPage);
    
    // 构建 HTML
    let html = `
        <style>
            .gallery-container {
                font-family: var(--font-interface);
            }
            .filter-bar {
                background: var(--background-secondary);
                padding: 16px;
                border-radius: 12px;
                margin-bottom: 20px;
                display: flex;
                flex-wrap: wrap;
                gap: 20px;
                align-items: flex-end;
            }
            .filter-group {
                display: flex;
                flex-direction: column;
                gap: 6px;
            }
            .filter-group label {
                font-size: 0.8em;
                font-weight: 600;
                color: var(--text-muted);
            }
            .filter-buttons {
                display: flex;
                flex-wrap: wrap;
                gap: 6px;
            }
            .filter-btn {
                padding: 4px 12px;
                border: 1px solid var(--background-modifier-border);
                border-radius: 20px;
                background: var(--interactive-normal);
                cursor: pointer;
                font-size: 0.85em;
                transition: 0.2s;
            }
            .filter-btn.active {
                background: var(--interactive-accent);
                color: white;
                border-color: var(--interactive-accent);
            }
            .rating-slider {
                display: flex;
                gap: 12px;
                align-items: center;
            }
            .rating-slider input {
                width: 180px;
            }
            .date-input {
                padding: 4px 8px;
                border-radius: 6px;
                border: 1px solid var(--background-modifier-border);
                background: var(--background-primary);
            }
            .cards-grid {
                display: grid;
                grid-template-columns: repeat(auto-fill, minmax(240px, 1fr));
                gap: 20px;
                margin: 20px 0;
            }
            .card {
                background: var(--background-primary-alt);
                border-radius: 16px;
                overflow: hidden;
                transition: transform 0.2s, box-shadow 0.2s;
                border: 1px solid var(--background-modifier-border);
            }
            .card:hover {
                transform: translateY(-4px);
                box-shadow: 0 8px 20px rgba(0,0,0,0.1);
            }
            .card-cover {
                aspect-ratio: 2 / 3;
                background: var(--background-modifier-border);
                display: flex;
                align-items: center;
                justify-content: center;
                font-size: 2em;
                color: var(--text-muted);
            }
            .card-cover img {
                width: 100%;
                height: 100%;
                object-fit: cover;
            }
            .card-info {
                padding: 12px;
            }
            .card-title {
                font-weight: 700;
                font-size: 1.05em;
                margin-bottom: 6px;
                white-space: nowrap;
                overflow: hidden;
                text-overflow: ellipsis;
            }
            .card-title a {
                text-decoration: none;
                color: var(--text-accent);
            }
            .card-author {
                font-size: 0.85em;
                color: var(--text-muted);
                margin-bottom: 8px;
            }
            .card-meta {
                display: flex;
                justify-content: space-between;
                align-items: center;
                font-size: 0.85em;
                margin-top: 8px;
            }
            .rating {
                background: var(--interactive-accent);
                color: white;
                padding: 2px 8px;
                border-radius: 20px;
                font-weight: 600;
            }
            .genre {
                background: var(--background-modifier-border);
                padding: 2px 8px;
                border-radius: 20px;
            }
            .pagination {
                display: flex;
                justify-content: center;
                gap: 12px;
                margin-top: 24px;
            }
            .page-btn {
                padding: 6px 14px;
                border: 1px solid var(--background-modifier-border);
                border-radius: 8px;
                background: var(--interactive-normal);
                cursor: pointer;
            }
            .result-info {
                margin: 12px 0;
                padding: 8px 12px;
                background: var(--background-secondary);
                border-radius: 8px;
                font-size: 0.9em;
            }
        </style>
        
        <div class="gallery-container">
            <!-- 筛选栏 -->
            <div class="filter-bar">
                <div class="filter-group">
                    <label>📚 类型</label>
                    <div class="filter-buttons" id="genre-buttons">
                        ${allGenres.map(g => `<button class="filter-btn ${currentGenre === g ? 'active' : ''}" data-genre="${g}">${g}</button>`).join('')}
                    </div>
                </div>
                
                <div class="filter-group">
                    <label>⭐ 评分范围</label>
                    <div class="rating-slider">
                        <span>${minRating}</span>
                        <input type="range" id="minRating" min="0" max="10" step="0.5" value="${minRating}">
                        <span>～</span>
                        <input type="range" id="maxRating" min="0" max="10" step="0.5" value="${maxRating}">
                        <span>${maxRating}</span>
                    </div>
                </div>
                
                <div class="filter-group">
                    <label>📅 添加日期</label>
                    <div class="filter-buttons" id="date-buttons">
                        <button class="filter-btn ${currentDateFilter === 'all' ? 'active' : ''}" data-date="all">全部</button>
                        <button class="filter-btn ${currentDateFilter === 'week' ? 'active' : ''}" data-date="week">最近一周</button>
                        <button class="filter-btn ${currentDateFilter === 'month' ? 'active' : ''}" data-date="month">最近一月</button>
                        <button class="filter-btn ${currentDateFilter === 'year' ? 'active' : ''}" data-date="year">最近一年</button>
                        <button class="filter-btn ${currentDateFilter === 'custom' ? 'active' : ''}" data-date="custom">自定义</button>
                    </div>
                    ${currentDateFilter === 'custom' ? `
                    <div style="display: flex; gap: 8px; margin-top: 6px;">
                        <input type="date" id="custom-start" class="date-input">
                        <span>至</span>
                        <input type="date" id="custom-end" class="date-input">
                        <button id="apply-custom" class="filter-btn">应用</button>
                    </div>
                    ` : ''}
                </div>
            </div>
            
            <div class="result-info">
                找到 ${filtered.length} 本小说 · 第 ${currentPage} / ${totalPages || 1} 页
            </div>
            
            <div class="cards-grid">
                ${pageData.map(book => `
                    <div class="card">
                        <div class="card-cover">
                            ${config.showCover && book.cover ? 
                                `<img src="${book.cover}" alt="封面" loading="lazy">` : 
                                `<span>${config.placeholderImage}</span>`
                            }
                        </div>
                        <div class="card-info">
                            <div class="card-title">${book.file.link}</div>
                            <div class="card-author">${book.author}</div>
                            <div class="card-meta">
                                <span class="rating">rating ${book.rating}</span>
                                <span class="genre">${book.genre}</span>
                            </div>
                        </div>
                    </div>
                `).join('')}
            </div>
            
            <div class="pagination">
                <button class="page-btn" id="prev-page" ${currentPage === 1 ? 'disabled' : ''}>← 上一页</button>
                <span>第 ${currentPage} / ${totalPages || 1} 页</span>
                <button class="page-btn" id="next-page" ${currentPage === totalPages || totalPages === 0 ? 'disabled' : ''}>下一页 →</button>
            </div>
        </div>
    `;
    
    dv.container.innerHTML = html;
    
    // 绑定事件
    const container = dv.container.querySelector('.gallery-container');
    if (!container) return;
    
    // 类型筛选
    container.querySelectorAll('#genre-buttons .filter-btn').forEach(btn => {
        btn.addEventListener('click', () => {
            currentGenre = btn.getAttribute('data-genre');
            currentPage = 1;
            renderGallery();
        });
    });
    
    // 评分滑块
    const minSlider = container.querySelector('#minRating');
    const maxSlider = container.querySelector('#maxRating');
    if (minSlider && maxSlider) {
        const updateRating = () => {
            minRating = parseFloat(minSlider.value);
            maxRating = parseFloat(maxSlider.value);
            if (minRating > maxRating) {
                [minRating, maxRating] = [maxRating, minRating];
                minSlider.value = minRating;
                maxSlider.value = maxRating;
            }
            currentPage = 1;
            renderGallery();
        };
        minSlider.addEventListener('input', updateRating);
        maxSlider.addEventListener('input', updateRating);
    }
    
    // 日期筛选
    container.querySelectorAll('#date-buttons .filter-btn').forEach(btn => {
        btn.addEventListener('click', () => {
            currentDateFilter = btn.getAttribute('data-date');
            if (currentDateFilter !== 'custom') {
                customStart = customEnd = null;
            }
            currentPage = 1;
            renderGallery();
        });
    });
    
    // 自定义日期
    const applyCustom = container.querySelector('#apply-custom');
    if (applyCustom) {
        applyCustom.addEventListener('click', () => {
            const startInput = container.querySelector('#custom-start');
            const endInput = container.querySelector('#custom-end');
            if (startInput && endInput && startInput.value && endInput.value) {
                customStart = dv.date(startInput.value);
                customEnd = dv.date(endInput.value);
                currentPage = 1;
                renderGallery();
            }
        });
    }
    
    // 分页
    const prevBtn = container.querySelector('#prev-page');
    const nextBtn = container.querySelector('#next-page');
    if (prevBtn) prevBtn.addEventListener('click', () => { if (currentPage > 1) { currentPage--; renderGallery(); } });
    if (nextBtn) nextBtn.addEventListener('click', () => { if (currentPage < totalPages) { currentPage++; renderGallery(); } });
}

renderGallery();
```
